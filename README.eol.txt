Copy/paste this configuration to your SVN client local
configuration file:



[miscellany]
enable-auto-props = yes

[auto-props]
*.fs     = svn:eol-style=native
*.fs7    = svn:eol-style=native
*.fsi    = svn:eol-style=native
*.txt    = svn:eol-style=native
*.tex    = svn:eol-style=native
*.bib    = svn:eol-style=native
Makefile = svn:eol-style=native
README   = svn:eol-style=native
